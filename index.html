My Engineering Projects

Self-Teaching Drum Set

This was my high school final project where I developed an electronic drum set which would play sounds, record and display each drum that was hit via MIDI, as well as having LED sequences which would help the user learn how to play the drums. The frame of the drums were made from PVC material and the drums were particle board, cnc cut from a design. I used piezo electric sensors to run via analog to the Arduino controller which then connected to the computer's serial via USB to be converted from analog data from the Arduino to MIDI. I used software called "Ardrumo" which takes the serial data from the Arduino and converts it to MIDI. The MIDI data then interfaces with "Garageband" which output the sounds of each drum. At the same time, I created code that ran in parallel with the sensors and the LED's on each drum so when each piezo sensor was hit, the LED would turn on for that specific drum. For the self teaching aspect, I created a few demos of basic beats and drum fills. I developed code where the LEDS would blink twice and then the third time is where the user would try to mimic the beat and it would record the data via the MIDI interface. My current method is programming the individual beats based on time delays. However, in the future, I would like to be able to play a beat on the drums and it would save the data to later be used as an alternate LED sequences for the learning aspect.

//LED DRUMSET CODE

#define LEDPIN     13     // status LED pin
#define PIEZOTHRESHOLD 5  // analog threshold for piezo sensing
#define PADNUM 6          // number of pads


const int sensorPin0 = A0;
const int sensorPin1 = A1;
const int sensorPin2 = A2;
const int sensorPin3 = A3;
const int sensorPin4 = A4;
const int sensorPin5 = A5;

const int ledPin12 = 12;
const int ledPin11 = 11;
const int ledPin10 = 10;
const int ledPin9 = 9;
const int ledPin8 = 8;
const int ledPin7 = 7;

const int threshold = 5;

int val;

void setup() 
{
  pinMode(LEDPIN, OUTPUT);
  Serial.begin(57600);   // set serial output rate
  pinMode(ledPin12, OUTPUT);
  pinMode(ledPin11, OUTPUT);
  pinMode(ledPin10, OUTPUT);
  pinMode(ledPin9, OUTPUT);
  pinMode(ledPin8, OUTPUT);
  pinMode(ledPin7, OUTPUT);
}

void loop() 
{

{
  int val0 = analogRead(sensorPin0);
  int val1 = analogRead(sensorPin1);
  int val2 = analogRead(sensorPin2);
  int val3 = analogRead(sensorPin3);
  int val4 = analogRead(sensorPin4);
  int val5 = analogRead(sensorPin5);
  if (val0 >= threshold)
 
  
  {
    digitalWrite(ledPin12, HIGH);
    delay(1);
    digitalWrite(ledPin12, LOW);
  }
  else
    digitalWrite(ledPin12, LOW);


if (val1 >= threshold)
{
  digitalWrite(ledPin11, HIGH);
  delay(1);
  digitalWrite(ledPin11, LOW);
}
else
  digitalWrite(ledPin11, LOW);



if (val2 >= threshold)
{
  digitalWrite(ledPin10, HIGH);
  delay(1);
  digitalWrite(ledPin10, LOW);
}
else
  digitalWrite(ledPin10, LOW);



if (val3 >= threshold)
{
  digitalWrite(ledPin9, HIGH);
  delay(1);
  digitalWrite(ledPin9, LOW);
}
else
  digitalWrite(ledPin9, LOW);



if (val4 >= threshold)
{
  digitalWrite(ledPin8, HIGH);
  delay(1);
  digitalWrite(ledPin8, LOW);
}
else
  digitalWrite(ledPin8, LOW);



if (val5 >= threshold)
{
  digitalWrite(ledPin7, HIGH);
  delay(1);
  digitalWrite(ledPin7, LOW);
}
else
  digitalWrite(ledPin7, LOW);
}
  
  
  for(int i = 0; i < PADNUM; i++) {
    val = analogRead(i);
    if( val >= PIEZOTHRESHOLD ) {
      digitalWrite(LEDPIN,HIGH);  // indicate we're sending MIDI data
      Serial.print(i);
      Serial.print(",");
      Serial.print(val);
      Serial.println();
      digitalWrite(LEDPIN,LOW);
    }
  }
}
